determCompl(fa):
    detFa <- initFa()    // Allocation...
    add(1,detFa.init)
    detFa.transTable[1][0] = fa.init
    while(state):
        for each char:
            detFa.transTable[state][char] = transitions(fa,state,char)
            if(notExistingState(detFa.transTable[state][char])):
                createState(detFa.transTable[state][char])
    for each state:
        for each t in fa.term:
            if fa.term[t] in detFa.transTable[state]
                add(state,detFa.term)
    if(notComplete(detFa)):
        complete(detFa)
    return detFa
